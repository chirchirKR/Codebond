A function that arrange cases e.g "deRiCk" to "Derick"


/** This function
  * Returns a the same parameter if not type of a string
  * Trims the start of a word 
  * Returns an arranged Case 
**/
function _Tt(word) {
  if (typeof word !== "string") return word;
  try {
    word = word.trimStart().toLowerCase();
    let __start__ = word.slice(0,1);
    let __end__ = word.slice(1, word.length);
    return (__start__.toUpperCase() + __end__);
  } catch(err) {
    console.error(err);
    return word;
  }
}

console.log(_Tt("derICk")); // Derick

Get users age automatically by just getting their date of birth


let user = {
  name: "",
  username: "",
  email: "",
  joined: "",
  age: "",
  dob(m) {
    const _yr = 31_536_000_000;
    const _now = Date.now();
    const _user = new Date(m).getTime();
    const _dob = Math.trunc(((_now - _user) / _yr));
    this.age = _dob;
  }
}

let [name, username, email, joined, _dob] = ["Derick", "@_derick7", "example@gmail.com", Date.now(),  "2004-12-7"];

user.name = name, user.username = username, user.email = email, user.joined = joined, user.dob(_dob);

console.log(user.age); // 20

Get Time in AM or PM using Javascript



let d = new Date(), h = d.getHours(), min = d.getMinutes(), mdf = '', hm = h % 12, str_hr = hm;

(min < 10) ? mdf = '0' : mdf = '';
let show = ` : ${mdf}${min} `;
let mid = "PM";

if (h > 0 && h < 12) {
  str_hr = hm;
  mid = "AM";
}

else if (h > 12 && h < 24) {
  str_hr = hm;
  mid = "PM";
}

else if (h === 0) {
 str_hr = 12;
 mid = "AM";
}

else {
 str_hr = 12; mid = "PM"
}

show = str_hr + show + mid;

console.log(show);
